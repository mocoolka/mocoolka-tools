lib/log-tools.js

import winston from 'winston';
import { fileTools } from '../index.js';

/**
 * get log function
 * @param {string}root
 * @param {boolean}isInfo
 * @param {boolean}isDebug
 * @returns {Function}
 */
const getLogFunction = (root, isInfo=false, isDebug=false)=> {
  fileTools.createDirNotExist(root);
  console.log(root);
  winston.handleExceptions(new winston.transports.File(
    { filename: fileTools.path(root, 'exceptions.log') }));
  winston.configure({
    transports: [
      new (winston.transports.File)({
        filename: fileTools.path(root, 'warning.log'),
        level: 'warning',
      }),
   ],
  });
  if (isInfo) {
    winston.add(winston.transports.File, {
      name: 'info',
      filename: 'info.log',
      level: 'info',
    });
  }

  if (isDebug) {
    winston.add(winston.transports.Console, {
      level: 'debug',
    });
  }

  winston.setLevels(winston.config.syslog.levels);
  return winston.log;
};

export default  {
  getLogFunction,
};
