/** @module mocoolka-tools-config */
import {
  loggerTools, fileTools, globalTools, errorTools, objectTools,
  envTools, cldrTools, stringTools, osTools,
}  from '../index.js';

let appConfig;
let config = {
  moduleName: 'mocoolka-tools-config',
  fileName: 'app.mk-config.json',
  filePath: null,
};

/**
 * set config value.it will try read app.mk-config.json in root directory
 * @param {string} configName - node name in app.mk-config.json
 * @param {Object} defaultConfig - default config
 * @param {Object} [useConfig] - user define config
 * @return {Object} merge with defaultConfig appConfig userConfig
 */
const setConfig = (configName, defaultConfig, useConfig) => {
  let rootPath;
  if (configName === 'mocoolka-tools-file')
    rootPath = useConfig.rootDirectory;
  let config = objectTools.mergeAll(false, true, defaultConfig, getAppConfig(configName, rootPath), useConfig);
  globalTools.setGlobalItem(configName, config);
  return config;
};

/**
 * get config value
 * @param {string}configName - node name in root node
 * @param {string}itemName - property name in node .if value is null return node object
 * @return {null|Object|string}
 */
const getConfig = (configName, itemName) => {
  let config = globalTools.getGlobalItem(configName);
  if (config) {
    if (itemName) {
      if (config[itemName]) {
        return config[itemName];
      } else {
        return null;
      }
    }
  }

  return config;
};

const init = (rootPath)=> {

  let configPath = fileTools.path(rootPath, config.fileName);
  if (fileTools.fileExist(configPath)) {
    appConfig = fileTools.openFile(configPath);
  } else {
    appConfig = {};
  }

  setConfig(config.moduleName, config, { filePath: rootPath });
  return appConfig;
};

/**
 * get appConfig
 * @param {string} [nodeName] node name in appConfig
 * @param {string} [rootPath] root path contain file app.mk-config.json
 * @return {Object}
 */
const getAppConfig = (nodeName, rootPath) => {
  if (!appConfig) {
    init(rootPath);
  }

  if (nodeName) {
    if (appConfig && appConfig[nodeName]) {
      return appConfig[nodeName];
    } else
      return null;
  }

  return appConfig;

};

export default {
  setConfig,
  getAppConfig,
  getConfig,
  init,
  moduleName: config.moduleName,
};
